version: '3.8'

services:
  # MySQL Database
  mysql:
    image: mysql:8.0
    container_name: alshalawi-mysql
    restart: always
    environment:
      MYSQL_ROOT_PASSWORD: ${DB_ROOT_PASSWORD:-rootpassword}
      MYSQL_DATABASE: ${DB_NAME:-alshalawi_gold}
      MYSQL_USER: ${DB_USER:-alshalawi_user}
      MYSQL_PASSWORD: ${DB_PASSWORD:-userpassword}
    ports:
      - "3306:3306"
    volumes:
      - mysql_data:/var/lib/mysql
      - ./database/schema.sql:/docker-entrypoint-initdb.d/schema.sql
    networks:
      - alshalawi-network
    healthcheck:
      test: ["CMD", "mysqladmin", "ping", "-h", "localhost"]
      timeout: 20s
      retries: 10

  # Backend API
  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: alshalawi-backend
    restart: always
    environment:
      NODE_ENV: production
      PORT: 5000
      DB_HOST: mysql
      DB_PORT: 3306
      DB_NAME: ${DB_NAME:-alshalawi_gold}
      DB_USER: ${DB_USER:-alshalawi_user}
      DB_PASSWORD: ${DB_PASSWORD:-userpassword}
      JWT_SECRET: ${JWT_SECRET:-change_this_secret}
      UPLOAD_PATH: ./uploads
      MAX_FILE_SIZE: 5242880
      CORS_ORIGIN: ${CORS_ORIGIN:-http://localhost:3000}
    ports:
      - "5000:5000"
    volumes:
      - ./backend/uploads:/app/uploads
      - ./backend/logs:/app/logs
    depends_on:
      mysql:
        condition: service_healthy
    networks:
      - alshalawi-network
    healthcheck:
      test: ["CMD", "wget", "--quiet", "--tries=1", "--spider", "http://localhost:5000/health"]
      interval: 30s
      timeout: 10s
      retries: 3

  # Nginx Reverse Proxy
  nginx:
    image: nginx:alpine
    container_name: alshalawi-nginx
    restart: always
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf
      - ./frontend/dist:/usr/share/nginx/html
      - ./backend/uploads:/usr/share/nginx/html/uploads
      - ./nginx/ssl:/etc/nginx/ssl
    depends_on:
      - backend
    networks:
      - alshalawi-network

volumes:
  mysql_data:
    driver: local

networks:
  alshalawi-network:
    driver: bridge
